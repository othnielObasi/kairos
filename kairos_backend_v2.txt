import { NanopaymentsClient } from '@circle-fin/nanopayments';

const nanoClient = new NanopaymentsClient({
  apiKey: process.env.CIRCLE_API_KEY!,
});

export const NANO_AMOUNT = parseFloat(
  process.env.NANOPAYMENT_AMOUNT_USDC || '0.001'
);

export interface NanopaymentReceipt {
  eventName:   string;
  source?:     string;
  model?:      string;
  type?:       string;
  txHash:      string;
  amount:      number;
  confirmedAt: number;
}

// billEvent — fires a $0.001 Circle Nanopayment for every governance/compute event
// NEVER throws — billing failure returns a pending receipt so governance is never blocked
export async function billEvent(
  eventName: string,
  meta: { source?: string; model?: string; type?: string } = {}
): Promise<NanopaymentReceipt> {
  try {
    const payment = await nanoClient.createPayment({
      from:     process.env.AGENT_WALLET_ADDRESS!,
      to:       process.env.GOVERNANCE_BILLING_ADDRESS!,
      amount:   NANO_AMOUNT.toString(),
      currency: 'USDC',
      metadata: {
        eventName,
        product:  'Kairos',
        agentId:  process.env.AGENT_ID || 'kairos-1',
        chain:    'Arc Testnet',
        ...meta,
      },
    });

    return {
      eventName,
      ...meta,
      txHash:      payment.txHash,
      amount:      NANO_AMOUNT,
      confirmedAt: Date.now(),
    };
  } catch (err) {
    // CRITICAL: never block governance on a billing failure
    console.warn(`[Kairos] billEvent failed (${eventName}):`, err);
    return {
      eventName,
      ...meta,
      txHash:      'pending_' + Date.now(),
      amount:      NANO_AMOUNT,
      confirmedAt: Date.now(),
    };
  }
}

import { NanopaymentReceipt } from './nanopayments';

const STAGE_NAMES = [
  'Mandate', 'Oracle', 'Simulator',
  'Supervisory', 'Risk Router', 'LLM Reasoning', 'SAGE',
];

class BillingStore {
  // Track 1 — governance Nanopayments
  t1Events:    NanopaymentReceipt[] = [];
  stageCounts: number[]             = new Array(7).fill(0);
  stageSpend:  number[]             = new Array(7).fill(0);
  t1Spend = 0;

  // Track 2 — x402 data API payments
  t2Events:    NanopaymentReceipt[]                            = [];
  apiBreakdown: Record<string, { calls: number; spend: number }> = {};
  t2Spend = 0;

  // Track 3 — compute Nanopayments
  t3Events: NanopaymentReceipt[] = [];
  t3Spend = 0;

  // Totals
  totalTxns  = 0;
  totalSpend = 0;

  addGovernanceEvent(r: NanopaymentReceipt, stageIndex: number) {
    r.source = STAGE_NAMES[stageIndex];
    r.type   = 'governance';
    this.t1Events.unshift(r);
    if (this.t1Events.length > 100) this.t1Events.pop();
    this.stageCounts[stageIndex]++;
    this.stageSpend[stageIndex]  += r.amount;
    this.t1Spend    += r.amount;
    this.totalSpend += r.amount;
    this.totalTxns++;
  }

  addApiEvent(r: NanopaymentReceipt, sourceKey: string) {
    r.type = 'data';
    this.t2Events.unshift(r);
    if (this.t2Events.length > 100) this.t2Events.pop();
    if (!this.apiBreakdown[sourceKey])
      this.apiBreakdown[sourceKey] = { calls: 0, spend: 0 };
    this.apiBreakdown[sourceKey].calls++;
    this.apiBreakdown[sourceKey].spend += r.amount;
    this.t2Spend    += r.amount;
    this.totalSpend += r.amount;
    this.totalTxns++;
  }

  addComputeEvent(r: NanopaymentReceipt) {
    r.type = r.type || 'compute';
    this.t3Events.unshift(r);
    if (this.t3Events.length > 100) this.t3Events.pop();
    this.t3Spend    += r.amount;
    this.totalSpend += r.amount;
    this.totalTxns++;
  }

  toJSON() {
    return {
      // Track 1
      t1Events:    this.t1Events.slice(0, 20),
      stageCounts: this.stageCounts,
      stageSpend:  this.stageSpend,
      t1Spend:     this.t1Spend,
      // Track 2
      t2Events:    this.t2Events.slice(0, 20),
      apiBreakdown: this.apiBreakdown,
      t2Spend:     this.t2Spend,
      // Track 3
      t3Events:    this.t3Events.slice(0, 20),
      t3Spend:     this.t3Spend,
      // Totals
      totalTxns:   this.totalTxns,
      totalSpend:  this.totalSpend,
      // Meta
      chain:       'Arc Testnet',
      chainId:     5042002,
      product:     'Kairos',
    };
  }
}

export const billingStore = new BillingStore();

  return wallet;
}

// Returns an ethers signer backed by Circle Wallets
// EIP-1271 compatible — works with existing src/chain/eip1271.ts
export function getCircleSigner(provider: ethers.Provider): ethers.AbstractSigner {
  return new CircleWalletSigner(
    process.env.AGENT_WALLET_ADDRESS!,
    process.env.AGENT_WALLET_ID!,
    provider
  );
}

class CircleWalletSigner extends ethers.AbstractSigner {
  constructor(
    private readonly addr:     string,
    private readonly walletId: string,
    provider: ethers.Provider
  ) {
    super(provider);
  }

  async getAddress()               { return this.addr; }
  connect(p: ethers.Provider)      { return new CircleWalletSigner(this.addr, this.walletId, p); }

  async signMessage(msg: string | Uint8Array): Promise<string> {
    const r = await circleClient.signMessage({
      walletId: this.walletId,
      message:  typeof msg === 'string' ? msg : Buffer.from(msg).toString('hex'),
    });
    return r.data?.signature ?? '';
  }

  async signTransaction(tx: ethers.TransactionRequest): Promise<string> {
    const r = await circleClient.createTransaction({
      walletId:            this.walletId,
      contractAddress:     tx.to as string,
      abiFunctionSignature:'',
      abiParameters:       [],
      feeLevel:            'MEDIUM',
    });
    return r.data?.transaction?.txHash ?? '';
  }

  async signTypedData(
    domain: ethers.TypedDataDomain,
    types:  Record<string, ethers.TypedDataField[]>,
    value:  Record<string, unknown>
  ): Promise<string> {
    const r = await circleClient.signTypedData({
      walletId:    this.walletId,
      domain,
      types,
      primaryType: Object.keys(types).find(k => k !== 'EIP712Domain') || '',
      message:     value,
    });
    return r.data?.signature ?? '';
  }
}

// ═══════════════════════════════════════════════════════════════════════════════
// TWO-WALLET ARCHITECTURE
// ═══════════════════════════════════════════════════════════════════════════════
//
// Wallet A — Circle Wallets (governance)
//   Used for: Circle Nanopayments (governance stages + compute billing)
//             Trade settlement on Arc (Track 4)
//   Identity: AGENT_WALLET_ADDRESS / AGENT_WALLET_ID
//   Funded:   Via Circle Developer Console
//
// Wallet B — BIP-39 mnemonic wallet (data payments)
//   Used for: x402 data API payments via Circle Gateway (Track 2)
//             Signs EIP-712 TransferWithAuthorization per data call
//   Identity: OWS_MNEMONIC derived address
//   Funded:   From faucet.circle.com (select Arc Testnet) → approve → deposit
//   Setup:    node src/services/setup-gateway.mjs (run once)
//
// Why two wallets:
//   Circle Wallets handles MPC-based programmatic signing for governance events.
//   The x402 client (x402_client.mjs) needs a BIP-39 mnemonic to derive the
//   EIP-712 signing key for TransferWithAuthorization. These are architecturally
//   incompatible — Circle Wallets cannot be used as the x402 mnemonic signer.
// ═══════════════════════════════════════════════════════════════════════════════
